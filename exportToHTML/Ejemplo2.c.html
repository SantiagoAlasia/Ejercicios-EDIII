2    *    @brief    Configuracion del periferico DAC para transferencia M2P con LLI Simple e ISR via GPDMA 
3    */
4    
5    #include "LPC17xx.h"
6    #include "lpc17xx_gpdma.h"
7    
8    #define DAC_BUFFER_START    0x20080000    //Primera direccion de la WaveForm
9    #define NUM_WAVE_SAMPLES    382
10   volatile uint32_t *dac_samples = (volatile uint32_t *)DAC_BUFFER_START;
11   volatile bool DAC_FLAG = false;
12   volatile uint32_t DMA_CH0_TC = 0;
13   volatile uint32_t DMA_CH0_ERR = 0;
14   
15   void cfgDMA(void);
16   void loadDACwave(void);
17   
18   int main(void){
19       cfgDMA();
20   
21       while(1){
22           if(DAC_FLAG){
23               loadDACwave();
24           }
25       }
26       return 0;
27   }
28   void loadDACwave(void){
29       GPDMA_ChannelCmd(LPC_GPDMACH0, ENABLE);    //Inicializacion de envio de datos por canal 0 de DMA
30       NVIC_EnableIRQ(DMA_IRQn);    //Habilitacion de interrupcion por DMA cuando termina de transferir los datos
31       return;
32   }
33   void cfgDAC(void){
34       uint32_t counDAC = 65535;    //Numero de cuentas para el timer dedicado del DAC
35       DAC_CONVERTER_CFG_Type cfgDAC;
36   
37       cfgDAC.CNT_ENA = SET;    //Habilitacion de TIMER dedicado al DAC
38       cfgDAC.DMA_ENA = SET;    //Habilitacion de DAC por DMA
39   
40       DAC_Init(LPC_DAC);    //Inicializacion del DAC
41   
42       DAC_SetBias(LPC_DAC, DAC_MAX_CURRENT_350uA);
43   
44       DAC_ConfigDAConverterControl(LPC_DAC, &cfgDAC);    //Configuracion del DAC
45       DAC_SetDMATimeOut(LPC_DAC, countDAC);    //Establecimiento del valor de recarga para el timer dedicado
46       return;
47   }
48   void cfgDMA(void){
49       GPDMA_LLI_Type cfgDAC_LLI0_CH0;    //Estructura para lista enlazada para canal 0 del DMA
50       GPDMA_Channel_CFG_Type cfgDAC_DMA_CH0;    //Estructura de configuracion para canal 0 del DMA
51   
52       NVIC_DisableIRQ(DMA_IRQn);    //Deshabilitacion de interupcion por DMA
53       GPDMA_Init();    //Inicializacion del controlador GPDMA
54   
55       cfgDAC_LLI0_CH0.SrcAddr = (uint32_t)dac_samples; //Direccion de Fuente de datos (Memoria)
56       cfgDAC_LLI0_CH0.DstAddr = (uint32_t)&(LPC_DAC->DACR); //Direccion de destino de datos(Periferico)
57       cfgDAC_LLI0_CH0.NextLLI = (uint32_t)&cfgDAC_LLI0_CH0; //Proxima direccion de destino de datos: LLI Repetitiva
58       cfgDAC_LLI0_CH0.Control = (NUM_WAVE_SAMPLES << 0)      //Tamaño de la transferencia de la LLI
59                               | (2 << 18)                   //Ancho de la transferencia Fuente (32 bit)
60                               | (2 << 21)                   //Ancho de la transferencia Destino (32 bit)
61                               | (1 << 26)                   //SI=1: Con incremento de la direccion de Fuente
62                               & ~(1 << 27);                 //DI=0: Sin incremento de la direccion de Destino
63   
64   
65       cfgDAC_DMA_CH0.ChannelNum = 0;    //Seleccion del canal de transferencia
66       cfgDAC_DMA_CH0.SrcConn = 0;    //Periferico de fuente de datos (Sin Uso)
67       cfgDAC_DMA_CH0.DstConn = GPDMA_CONN_DAC;    //Periferico de destino de datos (DAC)
68       cfgDAC_DMA_CH0.SrcMemAddr = (uint32_t)dac_samples;    //Direccion de fuente de datos en memoria
69       cfgDAC_DMA_CH0.DstMemAddr = 0;    //Direccion de destino de datos en memoria(Sin Uso)
70       cfgDAC_DMA_CH0.TransferType = GPDMA_TRANSFERTYPE_M2P;    //Tipo de transferencia (Memoria a Periferico)
71       cfgDAC_DMA_CH0.TransferSize = NUM_WAVE_SAMPLES;    //Tamaño de transferencia de datos
72       cfgDAC_DMA_CH0.TransferWidth 0;    //Ancho de transferencia de datos (Sin Uso)
73       cfgDAC_DMA_CH0.DMALLI = (uint32_t)&cfgDAC_LLI0_CH0;    //Lista enlazada incial
74       GPDMA_Setup(&cfgDAC_DMA_CH0);    //Configuracion del canal 0 de DMA
75       return;
76   }
77   void DMA_IRQHandler(void){
78       if(GPDMA_IntGetStatus(GPDMA_STAT_INT, 0)){
79           if(GPDMA_Int_Get_Status(GPDMA_STAT_INTTC, 0)){    //Verifica interrupcion por cuenta terminal
80               GPDMA_ClearPending(GPDMA_STATCLR_INTTC, 0);    //Limpia la bandera de interrupcion por cuenta terminal
81           }
82           if(GPDMA_IntGetStatus(GPDMA_STAT_INTERR, 0)){    //Verifica Interrupcion por error terminal
83               GPDMA_ClearPending(GPDMA_STATCLR_INTERR, 7);    //Limpia la bandera de interrupcion por error terminal
84           }
85       }
86       return;
87   }
