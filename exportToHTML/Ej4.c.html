1    /* 
2    Programar el microcontrolador LPC1769 para que mediante su ADC digitalice dos señales analógicas cuyos anchos 
3    de bandas son de 10 Khz cada una. Los canales utilizados deben ser el 2 y el 4 y los datos deben ser guardados 
4    en dos regiones de memorias distintas que permitan contar con los últimos 20 datos de cada canal. 
5    Suponer una frecuencia de core cclk de 100 Mhz. El código debe estar debidamente comentado. 
6    */

8    #include "LPC17xx.h"

14   void cfgADC(void);

17   #define MUESTRAS 20
18   #define ADC_BUFFER2_START    0x2007C000
19   #define ADC_BUFFER4_END    0x2007C014
20   #define ADC_CH2_ADDR ((uint32_t)&(LPC_ADC->ADDR2))
21   #define ADC_CH4_ADDR ((uint32_t)&(LPC_ADC ->ADDR4))
22   volatile uint32_t* buffer_2 = (volatile uint32_t *)ADC_BUFFER2_START;
23   uint32_t buffer_4[MUESTRAS];  // Para el canal 4 del ADC
24   
25   int main(void){
26       cfgGPIO();
27       cfgADC();
28       cfgDMA();
29       while(1){};
30       return 0;
31   }
32   void cfgGPIO(void){
33       /*canal 2: P0.25 
34         canal 4: P1.30*/
35       PINSEL_CFG_Type pinCfgADC2;
36   
37       pinCfgADC2.portNum = PINSEL_PORT_0;
38       pinCfgADC2.pinNum  = PINSEL_PIN_25;
39       pinCfgADC2.funcNum = PINSEL_FUNC_1;
40       pinCfgADC2.pinMode = PINSEL_PINMODE_TRISTATE;
41       pinCfgADC2.OpenDrain = PINSEL_PINMODE_NORMAL;
42   
43       PINSEL_ConfigPin(&pinCfgADC2);
44   
45       PINSEL_CFG_Type pinCfgADC4;
46   
47       pinCfgADC4.portNum = PINSEL_PORT_1;
48       pinCfgADC4.pinNum  = PINSEL_PIN_30;
49       pinCfgADC4.funcNum = PINSEL_FUNC_3;
50       pinCfgADC4.pinMode = PINSEL_PINMODE_TRISTATE;
51       pinCfgADC4.openDrain = PINSEL_PINMODE_NORMAL;
52   
53       PINSEL_ConfigPin(&pinCfgADC4);
54       return;
55   }
56   void cfgADC(void){
57       ADC_Init(LPC_ADC, 30000); // fs = 30kHZ (30 muestras por segundo)
58       ADC_BurstCmd(ENABLE);
59       ADC_StartCmd(LPC_ADC, ADC_START_CONTINUOS);
60   
61       ADC_ChannelCmd(ADC_CHANNEL_2, ENABLE);
62       ADC_ChannelCmd(ADC_CHANNEL_4, ENABLE);
63   
64       ADC_IntConfig(LPC_ADC, ADC_ADINT2, DISABLE);
65       ADC_IntConfig(LPC_ADC, ADC_ADINT4, DISABLE);
66       return;
67   }
68   void cfgDMA(void){
69       /*Transferencia P2M*/
70       NVIC_DiableIRQn(ADC_IRQn);
71       GPDMA_Init();
72   
73       GPDMA_Channel_CFG_Type cfgDMA_2;
74       GPDMA_Channel_CFG_Type cfgDMA_4;
75   
76       GPDMA_LLI_Type DMA_2_LLI0;
77       GPDMA_LLI_Type DMA_4_LLI0;
78   
79       //Canal 2
80       DMA_2_LLI0.SrcAddr = (uint32_t)&(LPC_ADC->ADDR2);
81       DMA_2_LLI0.DstAddr = (uint32_t)buffer_2;
82       DMA_2_LLI0.NextLLI = (uint32_t)&DMA_2_LLI0; // Circular
83       DMA_2_LLI0.Control = (MUESTRAS)|
84                      (2 << 18)| // Transfer width (src)
85                      (2 << 21)& // Transfer width (dst)
86                      ~(0<<26) | // SI -> NO incremento fuente
87                      (1<<27) | // DI -> SI incremento destino
88                      (1<<31); // Terminal Count Interrupt Enable
89   
90       cfgDMA_2.ChannelNum = 0;
91       cfgDMA_2.TransferSize = MUESTRAS;
92       cfgDMA_2.TransferWidth = GPDMA_WIDTH_WORD;
93       cfgDMA_2.SrcMemAddr = 0;
94       cfgDMA_2.DstMemAddr = (uint32_t)buffer_2;
95       cfgDMA_2.TransferType = GPDMA_TRANSFERTYPE_P2M;
96       cfgDMA_2.SrcConn = GPDMA_CONN_ADC;
97       cfgDMA_2.DstConn = 0;
98       cfgDMA_2.DMALLI =(uint32_t)&DMA_2_LLI0;
99   
100      GPDMA_Setup(&cfgDMA_2);
101      GPDMA_ChannelCmd(0, ENABLE);
102  
103      //Canal 4
104      DMA_4_LLI0.SrcAddr = (uint32_t)&(LPC_ADC->ADDR4);
105      DMA_4_LLI0.DstAddr = (uint32_t)buffer_4;
106      DMA_4_LLI0.NextLLI = (uint32_t)&DMA_4_LLI0; // Circular
107      DMA_2_LLI0.Control = (MUESTRAS)|
108                     (2 << 18)| // Transfer width (src)
109                     (2 << 21)& // Transfer width (dst)
110                     ~(0<<26) | // SI -> NO incremento fuente
111                     (1<<27) | // DI -> SI incremento destino
112                     (1<<31); // Terminal Count Interrupt Enable
113  
114      cfgDMA_4.ChannelNum = 1;
115      cfgDMA_4.TransferSize = MUESTRAS;
116      cfgDMA_4.TransferWidth = GPDMA_WIDTH_WORD;
117      cfgDMA_4.SrcMemAddr = 0;
118      cfgDMA_4.DstMemAddr = (uint32_t)buffer_4;
119      cfgDMA_4.TransferType = GPDMA_TRANSFERTYPE_P2M;
120      cfgDMA_4.SrcConn = GPDMA_CONN_ADC;
121      cfgDMA_4.DstConn = 0;
122      cfgDMA_4.DMALLI =(uint32_t)&DMA_4_LLI0;
123  
124      GPDMA_Setup (&cfgDMA_4);
125      GPDMA_ChannelCmd (1, ENABLE);
126      return;
127  }
128  void DMA_IRQHandler(void){
129  
130      volatile uint8_t flag_2 = 0;
131      volatile uint8_t flag_4 = 0;
132  
133      /*Canal 0: ADC del canal 2*/
134      if (GPDMA_IntGetStatus(GPDMA_STAT_INTTC, 0)) /*El canal termino de transferir 
135      la cantidad total de datos que se le pidio*/ {
136          GPDMA_ClearIntPending (GPDMA_STATCLR_INTTC, 0); // Limpia el flag
137       flag_2 = 1;
138      }
139  
140      /*Canal 1: ADC del canal 4*/
141      if(GPDMA_IntGetStatus(GPDMA_STAT_INTTC, 1)){
142          GPDMA_ClearIntPending (GPDMA_STATCLR_INTTC,1); // Limpia el flag
143          flag_4 = 1;
144      }
145  
146      /* En caso de error */
147      if (GPDMA_IntGetStatus(GPDMA_STAT_INTERR, 0)
148      || (GPDMA_IntGetStatus (GPDMA_STAT_INTERR,1))) {
149          GPDMA_ClearIntPending(GPDMA_STATCLR_INTERR, 0);
150          GPDMA_ClearIntPending(GPDMA_STATCLR_INTERR, 1);
151      }
152  }
